#include<iostream>
#include<string>
using namespace std;

template <class T>
class DLinkedList
{
  public:
        class Node;

  protected:
        Node *head;
        Node *tail;
        int count;
  public:
        DLinkedList();
        ~DLinkedList();
        void add(const T &e);
        void add(int index, const T &e);
        int size();
        bool empty();
        T get(int index);
        void set(int index, const T &e);
        int indexOf(const T &item);
        bool contains(const T &item);
        T removeAt(int index);
        bool removeItem(const T &item);
        void clear();
        string toString();
  public:
        class Node
        {
          private:
                T data;
                Node *next;
                Node *previous;
                friend class DLinkedList<T>;

          public:
                Node()
                {
                        this->previous = NULL;
                        this->next = NULL;
                }
                Node(const T &data)
                {
                        this->data = data;
                        this->previous = NULL;
                        this->next = NULL;
                }
        };
};

template <class T>
DLinkedList<T>::DLinkedList()
{
        head = NULL;
        tail = NULL;
}

template <class T>
DLinkedList<T>::~DLinkedList()
{
        head = NULL;
        tail = NULL;
}

// function to add elements to the linked list
template <class T>
void DLinkedList<T>::add(const T &e)
{
        Node *n =new Node(e);
        if(head==NULL)
        {
                head = tail = n;
        }
        else
        {
                tail->next = n;
                n->previous = tail;
                tail = n;
        }
}

// function to delete a element at a given index
template <class T>
T DLinkedList<T>::removeAt(int index)
{
        
        Node *t = head;
        
        if(index == 0)
        {
                head = head->next;
        }
        
        else
        {
                while(--index)
                {
                        t = t->next;
                }
                Node *temp = t;
                t->next = t->next->next;
                
                return temp->data;
        }
        
}

//function to remove a given item
template <class T>
bool DLinkedList<T>::removeItem(const T& item)
{
        Node *t = head;
        
        // traversing through the list
        while(t!=NULL)
        {
                // checking if data is found then deleting it
                if(t->data == item)
                {
                        Node* temp = t;
                        t = t->previous;
                        t = t->next->next;
                        
                        temp = NULL;
                        delete temp;
                        return true;
                }
        }
        
        // if element is not found then returning false
        return false;
}

// function to clear all the elements
template <class T>
void DLinkedList<T>::clear()
{
        head = NULL;
        tail = NULL;
}

template <class T>
string DLinkedList<T>::toString()
{
        string s = "[";
        Node *t = head;
        
        while(t!=NULL)
        {
                if(t->next == NULL)
                {
                        s += to_string(t->data);
                }
                else
                {
                        s += to_string(t->data) + ",";
                }
                
                t = t->next;
        }
        
        s += "]";
        return s;
}
